390 void handle_measurement(void) {
391  /* Start measurement if time has passed or user requested */
392  if (1 == flags.alarm || flags.refreshRequest) {
393  if (!start_soil_moisture_measurement(&hadc1, &hdma_adc1,
394  &sensor_readings.soilMoistureRaw)) {
395  Error_Handler();
396  }
397  flags.alarm = 0;
398  }
399  if(flags.soilMeasurementComplete){
400  sensor_readings.soilMoisturePercent = get_normalized_moisture_level(sensor_readings.soilMoistureRaw);
401  flags.soilMeasurementComplete = 0;
402  if(DHT11_get_data_from_sensor(&sensor_readings.dht11SensorData,&config) != DHT11_OK){
403  sensor_readings.dht11SensorData.Humidity = 0; /* If measurement was faulty alarm user*/
404  sensor_readings.dht11SensorData.Temperature = 0;
405  }
406  if(flags.refreshRequest){ /*Send data if user made a request*/
407  BLUETOOTH_INTERFACE_send_data((char *)out_buffer, sensor_readings);
408  HAL_UART_Transmit(&huart2, out_buffer,sizeof(out_buffer),HAL_MAX_DELAY);
409  __HAL_UART_FLUSH_DRREGISTER(&huart2);
410  flags.refreshRequest = 0;
411  return;
412  }
413  /*Start waterinng if water level is below set level */
414  if(sensor_readings.soilMoisturePercent < pump.water_level){
415  WATER_PUMP_start_watering(&pump);
416  }
417  /*Start next alarm*/
418  ALARM_RTC_set_alarm(&alarm_struct);
419 
420  }
